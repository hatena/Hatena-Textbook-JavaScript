fetch の使用例

fetch(url)
    .then(res => res.text())    // .json() や .blob() もある
    .then(r => console.log(r));

// いろいろオプション書いたり
var headers = new Header();
fetch(url, {
    method  : 'POST',
    headers : headers,
    cache   : 'no-cache',
});

// 途中でキャンセルできる  
var fetching = fetch(url);
abortButton.addEventListener('click', () => {
    fetching.then(res => res.body.cancel());
});
